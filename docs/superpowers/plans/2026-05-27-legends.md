# Legends Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 27 "Legend" stickers from PDF pages 100-101 of the Panini FWC2026 album. Users can collect them in a dedicated `/album/legends` section, AND optionally toggle a per-slot view in the team page to "paste the legend over" the normal player sticker (e.g., show Messi's Legend instead of ARG17 in the Argentina team page). Toggle persists in DB.

**Architecture:** Enfoque A — add `linked_sticker_id` (self-ref FK) to `stickers` + `display_variant` text column to `user_stickers`. 25 legends linked to player slots + 2 stand-alone (Neymar, Gilberto Mora). Reuses existing `SpecialSectionPage` for the new section. New `LegendToggle` client component shown over normal slots when user owns the linked legend. Render-time fallback rule prevents spoiler if user trades the legend away.

**Tech Stack:** Next.js 15 App Router · Supabase (Postgres + RLS) · Tailwind 4 · TypeScript · pnpm · PyMuPDF + Pillow for PDF extraction · `postgres` driver for verify scripts.

**Source spec:** [docs/superpowers/specs/2026-05-27-legends-design.md](../specs/2026-05-27-legends-design.md)

**Project rules (always respected):**
- `feedback-no-data-loss` — every catalog change is UPSERT, never DELETE. FK on `user_stickers.sticker_id` is `ON DELETE RESTRICT`. The new `linked_sticker_id` FK is also `ON DELETE RESTRICT`.
- `feedback-verify-not-assume` — every DB-touching task ends with a `test/` script that confirms it landed as expected.
- `feedback-test-folder` — scripts at `test/*.{ts,py}` are gitignored, used for ad-hoc verification.
- `feedback-push-default` — UI/UX commits are pushed immediately after each task.

**Verify scripts already present** (gitignored, in `test/`):

```sql
-- Schema for Legends (FWC2026 p100-101).
-- Adds two columns + one index. Pure ADD operations — zero data writes,
-- zero impact on existing rows. See spec
-- docs/superpowers/specs/2026-05-27-legends-design.md sección 1.

-- 1) Self-ref FK on stickers: legends point to the normal player slot
--    they cover. NULL for everything else. ON DELETE RESTRICT mirrors
--    user_stickers.sticker_id (set in 0028) — never cascade silently.
alter table public.stickers
  add column if not exists linked_sticker_id uuid
  references public.stickers(id) on delete restrict;

create index if not exists stickers_linked_idx
  on public.stickers(linked_sticker_id)
  where linked_sticker_id is not null;

-- 2) user_stickers.display_variant: per-user, per-slot override for the
--    team-view rendering. NULL = default (show normal); 'legend' = paste
--    the legend over. Only meaningful on rows where sticker_id is a
--    NORMAL player slot whose row in stickers has a legend linking to it.
alter table public.user_stickers
  add column if not exists display_variant text
  check (display_variant in ('normal', 'legend'));
```

```ts
 *   - stickers.linked_sticker_id column exists, type uuid, nullable
 *   - FK constraint exists with on-delete-restrict
 *   - stickers_linked_idx partial index exists
 *   - user_stickers.display_variant column exists, type text, with check

import postgres from "postgres";

async function main() {
  const sql = postgres(process.env.POSTGRES_URL_NON_POOLING!, {
    ssl: "require",
    max: 1,
  });

  const cols = await sql<
    Array<{ table_name: string; column_name: string; data_type: string; is_nullable: string }>
  >`
    select table_name, column_name, data_type, is_nullable
    from information_schema.columns
    where table_schema = 'public'
      and (
        (table_name = 'stickers' and column_name = 'linked_sticker_id')
        or (table_name = 'user_stickers' and column_name = 'display_variant')
      )
    order by table_name, column_name
  `;
  console.log("Columns:");
  console.table(cols);

  const fk = await sql<Array<{ constraint_name: string; delete_rule: string }>>`
    select rc.constraint_name, rc.delete_rule
    from information_schema.referential_constraints rc
    join information_schema.key_column_usage kcu
      on kcu.constraint_name = rc.constraint_name
    where kcu.table_schema = 'public'
      and kcu.table_name = 'stickers'
      and kcu.column_name = 'linked_sticker_id'
  `;
  console.log("FK linked_sticker_id:");
  console.table(fk);

  const idx = await sql<Array<{ indexname: string; indexdef: string }>>`
    select indexname, indexdef
    from pg_indexes
    where schemaname = 'public' and indexname = 'stickers_linked_idx'
  `;
  console.log("Index:");
  console.table(idx);

  const check = await sql<Array<{ constraint_name: string; check_clause: string }>>`
    select tc.constraint_name, cc.check_clause
    from information_schema.check_constraints cc
    join information_schema.constraint_column_usage ccu
      on ccu.constraint_name = cc.constraint_name
    join information_schema.table_constraints tc
      on tc.constraint_name = cc.constraint_name
    where ccu.table_schema = 'public'
      and ccu.table_name = 'user_stickers'
      and ccu.column_name = 'display_variant'
  `;
  console.log("Check constraint:");
  console.table(check);

  const allOk =
    cols.length === 2 &&
    fk.length === 1 &&
    fk[0].delete_rule === "RESTRICT" &&
    idx.length === 1 &&
    check.length === 1;
  console.log(allOk ? "\n✓ Migration 0029 verified" : "\n✗ Missing pieces");
```

```python
def trim_white(img, threshold=240):
    """Trim outer rows/cols that are nearly all white."""
    bbox = ImageChops.difference(
        img, Image.new(img.mode, img.size, (255, 255, 255))
    ).getbbox()
    return img.crop(bbox) if bbox else img


def crop_to_ratio(img, ratio):
    w, h = img.size
    cur = w / h
    if abs(cur - ratio) < 1e-3:
        return img
    if cur > ratio:
        new_w = int(round(h * ratio))
        x0 = (w - new_w) // 2
        return img.crop((x0, 0, x0 + new_w, h))
    new_h = int(round(w / ratio))
    y0 = (h - new_h) // 2
    return img.crop((0, y0, w, y0 + new_h))


def save_jpeg(img, path, quality=88):
    if img.mode != "RGB":
        img = img.convert("RGB")
    img.save(path, "JPEG", quality=quality, optimize=True, progressive=True)


def find_column_break(content, x_min_frac=0.5, x_max_frac=0.75, white_thr=240):
    """Find the vertical column boundary between horizontals and verticals.

    Scan a vertical strip between x_min_frac and x_max_frac of the content
    width, find the column with the most white pixels (the gap).
    """
    w, h = content.size
    px = content.load()
    best_x, best_white = None, -1
    for x in range(int(w * x_min_frac), int(w * x_max_frac)):
        white_count = sum(
            1 for y in range(h) if min(px[x, y][:3]) > white_thr
        )
        if white_count > best_white:
            best_white = white_count
            best_x = x
    return best_x


def slice_page(page_idx, page_num, doc):
    p = doc.load_page(page_idx)
    pix = p.get_pixmap(matrix=fitz.Matrix(5, 5), alpha=False)
    img = Image.frombytes("RGB", (pix.width, pix.height), pix.samples)
    content = trim_white(img)
    W, H = content.size
    # Save debug full-page
    content.save(f"{TMP}/p{page_num}-full.jpg", "JPEG", quality=80)

    # Find column break
    col_break = find_column_break(content)
    print(f"  p{page_num}: content {W}x{H}, col_break at x={col_break}")

    # Horizontals: 2 cols × 5 rows in left side
    horiz_w = col_break
    horiz_h = H
    row_h = horiz_h // 5
    half_w = horiz_w // 2

    # Verticals: 1 col × 4 rows in right side
    vert_w = W - col_break
    vert_h = H
    vert_row_h = vert_h // 4

    out = []
    for r in range(5):
        for c_label, x0, x1 in [
            ("L", 0, half_w),
            ("R", half_w, horiz_w),
        ]:
            slot = content.crop((x0, r * row_h, x1, (r + 1) * row_h))
            slot = trim_white(slot)
            slot = crop_to_ratio(slot, 560 / 420)
            slot = slot.resize((560, 420), Image.LANCZOS)
            key = (page_num, r, c_label)
            code = MAPPING.get(key, f"legtmp_p{page_num}_r{r}_c{c_label}")
            path = f"{DST}/{code}.jpg"
            save_jpeg(slot, path)
            out.append((key, code, slot.size, os.path.getsize(path) // 1024))
    for r in range(4):
        slot = content.crop((col_break, r * vert_row_h, W, (r + 1) * vert_row_h))
        slot = trim_white(slot)
        slot = crop_to_ratio(slot, 385 / 511)
        slot = slot.resize((385, 511), Image.LANCZOS)
        key = (page_num, r, "V")
        code = MAPPING.get(key, f"legtmp_p{page_num}_r{r}_cV")
        path = f"{DST}/{code}.jpg"
        save_jpeg(slot, path)
        out.append((key, code, slot.size, os.path.getsize(path) // 1024))
    return out
```

```ts
/** Cromos de una sección especial (apertura/historia/coca-cola/legends) */
export async function getStickersBySection(
  albumId: string,
  sectionKey: "apertura" | "historia" | "coca-cola" | "legends",
) {
  const supabase = await createClient();
  type Filter = {
    pageGte: number;
    pageLt: number;
    type: "legend" | null;
  };
  const filters: Record<typeof sectionKey, Filter> = {
    apertura:     { pageGte: 0,   pageLt: 100, type: null },
    historia:     { pageGte: 106, pageLt: 110, type: null },
    legends:      { pageGte: 100, pageLt: 102, type: "legend" },
    "coca-cola":  { pageGte: 110, pageLt: 120, type: null },
  };
  const f = filters[sectionKey];

  let q = supabase
    .from("stickers")
    .select("id, code, number, name, team, group_code, type, page")
    .eq("album_id", albumId)
    .is("group_code", null)
    .gte("page", f.pageGte)
    .lt("page", f.pageLt);
  if (f.type) q = q.eq("type", f.type);

  const { data: stickers } = await q
    .order("page", { ascending: true })
    .order("number", { ascending: true });
  return stickers ?? [];
}
```

### Task 11: Server action `setDisplayVariant`

**Files:**
- Modify: `src/app/(app)/album/actions.ts`

- [ ] **Step 1: Add the server action**

Edit `src/app/(app)/album/actions.ts`. Append AFTER the existing `setStickerQuantity` function:

```ts
/**
 * Set or clear the user's display variant for a slot in a team page.
 *
 * The slot is the NORMAL player sticker (e.g., ARG17). When variant='legend',
 * the team-view will render the legend's image/code/qty in place of the
 * normal — i.e., the user "pegó" the legend over the slot.
 *
 *   variant=null    → clear preference (defaults to normal)
 *   variant='normal' → explicit normal (same effect as null but persisted)
 *   variant='legend' → render the linked legend in this slot
 *
 * Validation:
 *   - slotStickerId must exist and NOT be type='legend'
 *   - If variant='legend': the slot must have a linked legend AND user must
 *     own ≥1 of that legend
 *
 * Mutation:
 *   - UPSERT user_stickers preserving quantity (never overwrite to 0)
 *   - If the user has no row for this slot, INSERT with quantity=0 +
 *     display_variant. Coherent with the album metaphor: "legend pegada
 *     sobre un slot vacío".
 */
export async function setDisplayVariant(
  slotStickerId: string,
  variant: "normal" | "legend" | null,
) {
  if (variant !== null && variant !== "normal" && variant !== "legend") {
    return { error: "Variante inválida" };
  }

  const supabase = await createClient();
  const {
    data: { user },
  } = await supabase.auth.getUser();
  if (!user) return { error: "No autenticado" };

  // 1) Slot exists and is not itself a legend
  const { data: slot } = await supabase
    .from("stickers")
    .select("id, type")
    .eq("id", slotStickerId)
    .maybeSingle();
  if (!slot) return { error: "Slot no existe" };
  if (slot.type === "legend") {
    return { error: "El slot no puede ser una legend" };
  }

  // 2) If setting 'legend': legend exists AND user owns it
  if (variant === "legend") {
    const { data: legend } = await supabase
      .from("stickers")
      .select("id")
      .eq("linked_sticker_id", slotStickerId)
      .eq("type", "legend")
      .maybeSingle();
    if (!legend) return { error: "Este slot no tiene legend asociada" };

    const { data: owned } = await supabase
      .from("user_stickers")
      .select("quantity")
      .eq("user_id", user.id)
      .eq("sticker_id", legend.id)
      .maybeSingle();
    if (!owned || (owned.quantity ?? 0) < 1) {
      return { error: "No posees esta legend" };
    }
  }

  // 3) UPSERT preserving quantity
  const { data: existing } = await supabase
    .from("user_stickers")
    .select("user_id")
    .eq("user_id", user.id)
    .eq("sticker_id", slotStickerId)
    .maybeSingle();

  if (existing) {
    const { error } = await supabase
      .from("user_stickers")
      .update({ display_variant: variant })
      .eq("user_id", user.id)
      .eq("sticker_id", slotStickerId);
    if (error) return { error: error.message };
  } else {
    const { error } = await supabase
      .from("user_stickers")
      .insert({
        user_id: user.id,
        sticker_id: slotStickerId,
        quantity: 0,
        display_variant: variant,
      });
    if (error) return { error: error.message };
  }

  revalidatePath("/album", "layout");
  return { success: true };
}
```

```tsx
function StickerCardSlot({
  s,
  qtyMap,
  horizontal,
  readOnly,
  legendQtyMap,

  s: SectionSticker;
  qtyMap: Map<string, number>;
  horizontal?: boolean;
  readOnly?: boolean;
  legendQtyMap?: Map<string, number>;
}) {
  // Render rule defensiva: solo respeta variant='legend' SI user posee la legend
  const effectiveVariant: "normal" | "legend" =
    s.displayVariant === "legend" && s.hasLegend ? "legend" : "normal";
  const showToggle =
    !readOnly && Boolean(s.hasLegend) && Boolean(s.legendStickerId) && Boolean(s.legendCode);

  const renderId = effectiveVariant === "legend" ? s.legendStickerId! : s.id;
  const renderQty =
    effectiveVariant === "legend"
      ? legendQtyMap?.get(s.legendStickerId!) ?? 0
      : qtyMap.get(s.id) ?? 0;
  const renderType: SectionSticker["type"] =
```
